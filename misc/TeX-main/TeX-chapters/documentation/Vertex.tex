Vertex idtype
idtype - used to identify individual vertices. Small primitive types are preferable.

\paragraph{Interface}


Construction

\begin{lstlisting}
                Vertex(           id_type);

                Vertex(     const Vertex&);

                Vertex(           Vertex&&);


const Vertex&   operator=(  const Vertex&);
\end{lstlisting}


Element accessors

\begin{lstlisting}
const id_type&     Id()           const;

      void         Id(   id_type);
\end{lstlisting}

Example:

\begin{lstlisting}
#include "graph-lib/graph.hpp"
#include <string>


int main(int argv, char** argc)
{
   Graph<int, UndirectedEdge> testGraph = { Vertex<int>(1),
                                            Vertex<int>(2),
                                            Vertex<int>(3) };
   return 0;
}

\end{lstlisting}
