Vertex idtype
idtype - used to identify individual vertices. Small primitive types are preferable.

\paragraph{Interface}


Construction

\begin{lstlisting}
                Edge(      const   Vertex<Data>&, 
                           const   Vertex<Data>&, 
                                   int);
		
                Edge(      const   Edge&);

                Edge(              Edge&&);


const Edge&	    operator=( const Edge&);
\end{lstlisting}


Element accessors

\begin{lstlisting}
const Vertex<Data>&     First()         const;

const Vertex<Data>&     Second()        const;

      int               Weight()        const;

      void              Weight(   int);
\end{lstlisting}

Usefull type aliases

\begin{lstlisting}
UndirectedEdge = Edge<id_type, false>;
DirectedEdge   = Edge<id_type, true>;
\end{lstlisting}

Example:

\begin{lstlisting}
#include "graph-lib/graph.hpp"
#include <string>


int main(int argv, char** argc)
{
   Graph<int, UndirectedEdge> testGraph = { Vertex<int>(1),
                                            Vertex<int>(2),
                                            Vertex<int>(3) };
   return 0;
}

\end{lstlisting}
